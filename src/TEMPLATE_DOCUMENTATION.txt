team.html
path: /teams/<name>/
context: {
    'members': (список из имен участников команды. НЕ объекты самих пользователей, только имена)
    'owner': (Является ли человек, просматривающий страницу владельцем этой команды? True False)
    'team': (объект текущей команды)
    'ad': (объект анкеты команды, если имеется. Иначе None)
    'isMember': (является ли текущий пользователь участником данной команды? True False)
    'canRequest': (может ли текущий пользователь отправить запрос в команду, владельцем которой является данный пользователь?)
}

user.html
path: /users/<name>/
context: {
    'user': (объект пользователя)
    'team': (объект команды, в которой состоит данный пользователь. Иначе None)
    'ad': (объект анкеты пользователя, если имеется. Иначе None)
    'owner': (ялвяется ли текущий пользователь владельцем этой страницы? True False)
    'invites': (все приглашения в команду для данного пользователя, если owner=True)
    'requests': (все запросы на присоединение в команду текущего пользователя если owner=True)
    'canInvite': (можно ли данного пользователя пригласить в команду?)
    'canKick': (можно ли данного пользователя выгнать из команды?)
}
